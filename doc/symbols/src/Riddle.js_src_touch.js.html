1 r(function() {
  2 
  3   var body = document.body,
  4       touchEvents = ["swipeLeft", "swipeRight"],
  5       touch = { x1: 0, x2: 0 };
  6 
  7   function search(node) {
  8     return 'tagName' in node ? node : node.parentNode;
  9   }
 10 
 11   function trigger(elem, event) {
 12     var e = document.createEvent("Event");
 13     e.initEvent(event, true, true);
 14     elem.dispatchEvent(e);
 15   }
 16 
 17   r(body).bind("touchstart", function(e) {
 18     touch.target = search(e.touches[0].target);
 19     touch.x1 = e.touches[0].pageX;
 20     //console.dir(e);
 21   });
 22 
 23   r(body).bind("touchmove", function(e) {
 24     touch.x2 = e.touches[0].pageX;
 25     //console.dir(e);
 26   });
 27 
 28   r(body).bind("touchend", function(e) {
 29     var delta = touch.x1 - touch.x2;
 30 
 31     if ( touch.x1 && touch.x2 && Math.abs(delta) > 30 ) {
 32       if ( delta > 0 ) {
 33         trigger(touch.target, "swipeLeft");
 34       }
 35       else {
 36         trigger(touch.target, "swipeRight");
 37       }
 38     }
 39 
 40     touch.target = null;
 41     touch.x1 = touch.x2 = 0;
 42   });
 43 
 44   touchEvents.forEach(function(event) {
 45     r.fn[event] = function(callback, useCapture) {
 46       this.bind(event, callback, useCapture);
 47     };
 48   });
 49 
 50 });
